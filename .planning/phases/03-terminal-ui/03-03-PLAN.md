```typescript
  // ─── live update via fs.watch ─────────────────────────────────────────────
  const callsFile = path.join(TRACER_DIR, 'sessions', sessionId, 'calls.jsonl');
  let pendingNew = 0;
  let atBottom = calls.length <= 1; // start at bottom if ≤1 call

  // Track whether user is at the last item
  function checkAtBottom(): boolean {
    const selected = (timeline as unknown as { selected: number }).selected;
    return selected >= calls.length - 1;
  }

  function highlightRow(index: number): void {
    // Temporarily set the item text to cyan-tagged version
    const original = timelineItem(calls[index]);
    const cyanItem = `{cyan-fg}${original.trim()}{/cyan-fg}`;
    timeline.setItem(index, cyanItem as unknown as blessed.Widgets.BlessedElement);
    screen.render();
    setTimeout(() => {
      if (index < calls.length) {
        timeline.setItem(index, timelineItem(calls[index]) as unknown as blessed.Widgets.BlessedElement);
        screen.render();
      }
    }, 1000);
  }

  function updateTimelineLabel(): void {
    if (pendingNew > 0) {
      (timeline as unknown as { setLabel: (s: string) => void }).setLabel(` Timeline  {cyan-fg}+${pendingNew} new{/cyan-fg} `);
    } else {
      (timeline as unknown as { setLabel: (s: string) => void }).setLabel(' Timeline ');
    }
  }

  let fsWatcher: fs.FSWatcher | null = null;
  try {
    fsWatcher = fs.watch(callsFile, { persistent: true }, () => {
      const newCalls = readCalls(sessionId);
      if (newCalls.length > calls.length) {
        const added = newCalls.slice(calls.length);
        const wasAtBottom = checkAtBottom();

        // Append to calls array
        calls.push(...added);

        // Add items to timeline list
        for (const c of added) {
          (timeline as unknown as { addItem: (s: string) => void }).addItem(timelineItem(c));
        }

        updateStatus();

        if (wasAtBottom) {
          // Auto-scroll to the newest call
          const newIndex = calls.length - 1;
          timeline.select(newIndex);
          selectCall(newIndex);
          pendingNew = 0;
        } else {
          // Accumulate pending count and update label
          pendingNew += added.length;
          updateTimelineLabel();
        }

        // Highlight each new row cyan for 1 second
        for (let i = calls.length - added.length; i < calls.length; i++) {
          highlightRow(i);
        }

        screen.render();
      }
    });
  } catch {
    // Session file may not exist yet (0 calls); watcher will be retried on next render cycle
  }

  // When user navigates to bottom, clear pending indicator
  timeline.on('keypress', () => {
    if (checkAtBottom() && pendingNew > 0) {
      pendingNew = 0;
      updateTimelineLabel();
      screen.render();
    }
  });

  // Cleanup watcher on quit
  const originalQuit = () => {
    if (fsWatcher) fsWatcher.close();
    screen.destroy();
    process.exit(0);
  };
  // Re-register quit to also close watcher
  screen.unkey(['q', 'C-c'], () => {});
  screen.key(['q', 'C-c'], originalQuit);
```

Ensure `path` is also imported: `import * as path from 'path';` (already present from 03-02).
