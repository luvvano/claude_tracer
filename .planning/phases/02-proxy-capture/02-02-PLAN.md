# Plan 02-02: Show & Diff CLI Commands

## Objective
Add `claude-tracer show`, `claude-tracer show <session_id>`, and `claude-tracer diff <session_id> <call_index>` commands to the CLI for inspecting session history and per-call prompt diffs in the terminal.

## Requirements Covered
CLI-01, CLI-02, CLI-03

## must_haves
- [ ] `claude-tracer show` (no args) lists all sessions sorted newest-first with id, date, call count, total tokens
- [ ] `claude-tracer show <session_id>` prints full call timeline with diff summaries per call
- [ ] `claude-tracer diff <session_id> <call_index>` prints the diff for a specific call
- [ ] Token totals read from `token_total` field written by 02-01
- [ ] All three commands handle missing session/call gracefully (error message + exit 1)
- [ ] Build passes: `npm run build` exits 0
- [ ] Manual smoke test of all three commands succeeds against a real session

```typescript
#!/usr/bin/env node
import { Command } from 'commander';
import * as child_process from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import { DaemonState, CallRecord } from './types';

const TRACER_DIR = path.join(os.homedir(), '.claude-tracer');
const PID_FILE = path.join(TRACER_DIR, 'daemon.pid');
const PROXY_SCRIPT = path.join(__dirname, 'proxy.js');
const SEP = '─'.repeat(49);

// ── helpers ──────────────────────────────────────────────────────────────────

function readState(): DaemonState | null {
  try {
    const raw = fs.readFileSync(PID_FILE, 'utf8');
    return JSON.parse(raw) as DaemonState;
  } catch {
    return null;
  }
}

function isRunning(pid: number): boolean {
  try {
    process.kill(pid, 0);
    return true;
  } catch {
    return false;
  }
}

function countCalls(sessionId: string): number {
  const file = path.join(TRACER_DIR, 'sessions', sessionId, 'calls.jsonl');
  try {
    const content = fs.readFileSync(file, 'utf8');
    return content.split('\n').filter(l => l.trim()).length;
  } catch {
    return 0;
  }
}

function readCalls(sessionId: string): CallRecord[] {
  const file = path.join(TRACER_DIR, 'sessions', sessionId, 'calls.jsonl');
  try {
    const content = fs.readFileSync(file, 'utf8');
    return content
      .split('\n')
      .filter(l => l.trim())
      .map(l => JSON.parse(l) as CallRecord);
  } catch {
    return [];
  }
}

function listSessions(): string[] {
  const sessionsDir = path.join(TRACER_DIR, 'sessions');
  try {
    return fs
      .readdirSync(sessionsDir)
      .filter(name => {
        try {
          return fs.statSync(path.join(sessionsDir, name)).isDirectory();
        } catch {
          return false;
        }
      })
      .sort()
      .reverse();
  } catch {
    return [];
  }
}

function fmtNumber(n: number): string {
  return n.toLocaleString('en-US');
}

function fmtTime(isoTs: string): string {
  // Returns HH:MM:SS
  try {
    const d = new Date(isoTs);
    return d.toTimeString().slice(0, 8);
  } catch {
    return isoTs;
  }
}

function fmtDate(isoTs: string): string {
  // Returns YYYY-MM-DD HH:MM
  try {
    const d = new Date(isoTs);
    return d.toISOString().slice(0, 16).replace('T', ' ');
  } catch {
    return isoTs;
  }
}

function diffEntryLine(entry: { role: string; content_summary: string; is_tool_use: boolean; tool_name?: string }): string {
  if (entry.is_tool_use && entry.tool_name) {
    // Extract tool input path/target if possible
    try {
      const inputMatch = entry.content_summary.match(/"input"\s*:\s*\{([^}]*)\}/);
      const pathMatch = entry.content_summary.match(/"(?:path|file_path|command)"\s*:\s*"([^"]+)"/);
      const target = pathMatch ? ` → ${pathMatch[1]}` : '';
      return `  + ${entry.role}: [tool_use: ${entry.tool_name}${target}]`;
    } catch {
      return `  + ${entry.role}: [tool_use: ${entry.tool_name}]`;
    }
  }

  // Detect tool_result
  if (entry.content_summary.includes('"type":"tool_result"') || entry.content_summary.includes('"type": "tool_result"')) {
    const lenMatch = entry.content_summary.match(/"content"\s*:\s*"([^"]{0,200})"/);
    const charCount = lenMatch ? lenMatch[1].length : '?';
    return `  + ${entry.role}: [tool_result: ${charCount} chars]`;
  }

  // Plain text — show first 60 chars
  const preview = entry.content_summary.replace(/\n/g, ' ').slice(0, 60);
  return `  + ${entry.role}: "${preview}"`;
}

// ── program ──────────────────────────────────────────────────────────────────

const program = new Command();

program
  .name('claude-tracer')
  .description('Transparent proxy for tracing Claude Code API calls')
  .version('0.1.0');

// ── start ────────────────────────────────────────────────────────────────────

program
  .command('start')
  .description('Start the proxy daemon on port 7749')
  .option('-f, --foreground', 'Run in foreground (do not detach)')
  .action((opts: { foreground?: boolean }) => {
    const existing = readState();
    if (existing && isRunning(existing.pid)) {
      console.log(`Already running (PID ${existing.pid}, session: ${existing.sessionId})`);
      process.exit(0);
    }

    if (opts.foreground) {
      require('./proxy').startProxy();
      return;
    }

    const child = child_process.spawn(process.execPath, [PROXY_SCRIPT], {
      detached: true,
      stdio: ['ignore', 'pipe', 'pipe'],
    });

    child.stdout?.on('data', (chunk: Buffer) => {
      process.stdout.write(chunk);
    });
    child.stderr?.on('data', (chunk: Buffer) => {
      process.stderr.write(chunk);
    });

    child.unref();

    setTimeout(() => {
      const state = readState();
      if (state && isRunning(state.pid)) {
        console.log(`\nProxy started. Set: export ANTHROPIC_BASE_URL=http://localhost:7749`);
        process.exit(0);
      } else {
        console.error('Proxy failed to start (no pid file written)');
        process.exit(1);
      }
    }, 1500);
  });

// ── stop ─────────────────────────────────────────────────────────────────────

program
  .command('stop')
  .description('Stop the proxy daemon')
  .action(() => {
    const state = readState();
    if (!state) {
      console.log('Not running.');
      process.exit(0);
    }
    if (!isRunning(state.pid)) {
      console.log('Not running (stale pid file). Cleaning up.');
      try { fs.unlinkSync(PID_FILE); } catch { /* ignore */ }
      process.exit(0);
    }
    try {
      process.kill(state.pid, 'SIGTERM');
      console.log(`Stopped PID ${state.pid} (session: ${state.sessionId})`);
      setTimeout(() => {
        if (fs.existsSync(PID_FILE)) {
          try { fs.unlinkSync(PID_FILE); } catch { /* ignore */ }
        }
        process.exit(0);
      }, 500);
    } catch (err) {
      console.error('Failed to stop:', (err as Error).message);
      process.exit(1);
    }
  });

// ── status ───────────────────────────────────────────────────────────────────

program
  .command('status')
  .description('Show proxy daemon status')
  .action(() => {
    const state = readState();
    if (!state || !isRunning(state.pid)) {
      console.log('Status: stopped');
      process.exit(0);
    }
    const calls = countCalls(state.sessionId);
    console.log(`Status:     running`);
    console.log(`PID:        ${state.pid}`);
    console.log(`Session ID: ${state.sessionId}`);
    console.log(`Port:       ${state.port}`);
    console.log(`Started:    ${state.startedAt}`);
    console.log(`Calls:      ${calls}`);
    console.log(`\nTo use: export ANTHROPIC_BASE_URL=http://localhost:${state.port}`);
    process.exit(0);
  });

// ── show ─────────────────────────────────────────────────────────────────────

program
  .command('show [session_id]')
  .description('List sessions or show call timeline for a session')
  .action((sessionId?: string) => {
    if (!sessionId) {
      // List all sessions
      const sessions = listSessions();
      if (sessions.length === 0) {
        console.log('No sessions found.');
        process.exit(0);
      }

      for (const sid of sessions) {
        const calls = readCalls(sid);
        const callCount = calls.length;
        const lastCall = calls[calls.length - 1];
        const totalTokens = lastCall?.token_total ?? calls.reduce((sum, c) => sum + (c.usage?.input_tokens ?? 0), 0);
        const firstTs = calls[0]?.ts ?? '';
        const dateStr = firstTs ? fmtDate(firstTs) : '(unknown)';
        const tokStr = fmtNumber(totalTokens);
        console.log(`${sid}  ${dateStr}  ${callCount} calls  ${tokStr} tokens`);
      }
      process.exit(0);
    }

    // Show call timeline for a specific session
    const calls = readCalls(sessionId);
    if (calls.length === 0) {
      console.error(`No calls found for session: ${sessionId}`);
      process.exit(1);
    }

    console.log(`Session: ${sessionId}`);
    console.log(SEP);

    let totalOutputTokens = 0;

    for (const call of calls) {
      const timeStr = fmtTime(call.ts);
      const inputTok = call.usage?.input_tokens ?? 0;
      const outputTok = call.usage?.output_tokens ?? 0;
      totalOutputTokens += outputTok;
      const durationSec = (call.duration_ms / 1000).toFixed(1);
      const prevTotal = (call.token_total ?? inputTok) - inputTok;
      const deltaStr = call.call_index > 0 ? `  (+${fmtNumber(inputTok)})` : '';

      console.log(`Call ${call.call_index}  ${timeStr}  ${call.model}  ${fmtNumber(call.token_total ?? inputTok)} tok  ${durationSec}s${deltaStr}`);

      if (call.diff && call.diff.length > 0) {
        for (const entry of call.diff) {
          console.log(diffEntryLine(entry));
        }
      } else if (call.call_index === 0 && call.messages && (call.messages as unknown[]).length > 0) {
        // First call: show the first user message as context
        const firstMsg = (call.messages as Record<string, unknown>[])[0];
        if (firstMsg) {
          const content = typeof firstMsg['content'] === 'string'
            ? firstMsg['content'].slice(0, 60)
            : JSON.stringify(firstMsg['content']).slice(0, 60);
          console.log(`  + ${firstMsg['role']}: "${content}"`);
        }
      }
    }

    const lastCall = calls[calls.length - 1];
    const totalTokens = lastCall?.token_total ?? 0;
    console.log(SEP);
    console.log(`Total: ${calls.length} calls | ${fmtNumber(totalTokens)} input tokens | ${fmtNumber(totalOutputTokens)} output tokens`);
    process.exit(0);
  });

// ── diff ─────────────────────────────────────────────────────────────────────

program
  .command('diff <session_id> <call_index>')
  .description('Show prompt diff for a specific call (what messages were added)')
  .action((sessionId: string, callIndexStr: string) => {
    const callIndex = parseInt(callIndexStr, 10);
    if (isNaN(callIndex) || callIndex < 0) {
      console.error(`Invalid call_index: ${callIndexStr}`);
      process.exit(1);
    }

    const calls = readCalls(sessionId);
    if (calls.length === 0) {
      console.error(`No calls found for session: ${sessionId}`);
      process.exit(1);
    }

    const call = calls.find(c => c.call_index === callIndex);
    if (!call) {
      console.error(`Call index ${callIndex} not found in session ${sessionId}`);
      process.exit(1);
    }

    const diff = call.diff ?? [];
    const totalMessages = (call.messages as unknown[]).length;
    const newCount = diff.length;
    const carriedCount = totalMessages - newCount;

    console.log(`Call ${callIndex} diff (${newCount} messages added):`);
    console.log(SEP);

    if (diff.length === 0) {
      console.log('  (no diff — this is the first call or diff data unavailable)');
    } else {
      for (const entry of diff) {
        console.log(`[+] ${entry.role} (index=${entry.index})`);
        if (entry.is_tool_use && entry.tool_name) {
          // Try to extract input fields for display
          const inputMatch = entry.content_summary.match(/"input"\s*:\s*(\{[^}]*\})/);
          const inputStr = inputMatch ? inputMatch[1] : '';
          console.log(`    content: [tool_use] ${entry.tool_name} ${inputStr}`);
        } else {
          const preview = entry.content_summary.replace(/\\n/g, '\n').slice(0, 200);
          console.log(`    content: ${preview}${entry.content_summary.length > 200 ? '...' : ''}`);
        }
        console.log('');
      }
    }

    console.log(SEP);
    console.log(`Full messages[]: ${totalMessages} total (${carriedCount} carried + ${newCount} new)`);
    process.exit(0);
  });

program.parse(process.argv);
```
